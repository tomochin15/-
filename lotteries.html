from __future__ import annotations import re from typing import Iterable, Tuple def clean(text: str) -> str: return re.sub(r"\s+", " ", text).strip() DT_PATTERNS = [ re.compile(r"(20\d{2}[/-]\d{1,2}[/-]\d{1,2}\s*\d{1,2}:\d{2})"), re.compile(r"(20\d{2}年\d{1,2}月\d{1,2}日\s*\d{1,2}:\d{2})"), re.compile(r"(\d{1,2}月\d{1,2}日\s*\d{1,2}:\d{2})"), ] def find_near_datetime(text: str, keywords: Iterable[str], fallback: str = "公式ページ確認") -> str: lines = [x.strip() for x in re.split(r"[。\n\r]+", text) if x.strip()] for line in lines: if any(k in line for k in keywords): for pat in DT_PATTERNS: m = pat.search(line) if m: return m.group(1) return fallback def extract_schedule(text: str) -> Tuple[str, str, str]: start = find_near_datetime(text, ["応募受付開始", "受付開始", "お申し込み受付開始", "販売開始", "開始"]) end = find_near_datetime(text, ["応募受付終了", "受付終了", "受付締切", "お申し込み受付終了", "応募締切", "締切"]) result = find_near_datetime(text, ["当選発表", "結果発表", "抽選結果", "当落発表"]) return start, end, result
